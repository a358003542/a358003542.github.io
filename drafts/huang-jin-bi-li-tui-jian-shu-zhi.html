黄金比例推荐数值

前言

在设计上似乎放你的按钮或者其他方块形状遵守黄金比例是一个不错的选择，所以编写下面的程序来试着找黄金比例的最推荐的数值。

import sys
import os


GoldenRatio = 1.61803399
HEIGHT_MAX = 2000

outfile = "GoldenRatioResult.txt"


def test(precision=3, file=sys.stdout):
    for height in range(1, HEIGHT_MAX):
        width = height * GoldenRatio
        string_width = str(width)

        dot_pre, dot_last = string_width.split(".")

        bool_conditions = []

        for i in range(0, precision):

            if (dot_last[i] == "0"):

                bool_conditions.append(True)
            else:

                bool_conditions.append(False)

        if (all(bool_conditions)):
            print(string_width, height, file=file)


def test2(precision=3, file=sys.stdout):

    for height in range(1, HEIGHT_MAX):

        # even
        if not (height % 2 == 0):
            continue

        width = height * GoldenRatio
        string_width = str(width)

        dot_pre, dot_last = string_width.split(".")

        if not (dot_pre[-1] in ["0", "2", "4", "6", "8"]):
            continue

        bool_conditions = []

        for i in range(0, precision):
            if (dot_last[i] == "0"):
                bool_conditions.append(True)
            else:
                bool_conditions.append(False)

        if (all(bool_conditions)):
            print(string_width, height, file=file)


def main():
    if (os.path.exists(outfile)):
        os.remove(outfile)

    for i in range(4, 0, -1):
        with open(outfile, 'at+', encoding='utf8') as f:
            print(f"------precision {i}------", file=f)
            test(i, file=f)

    for i in range(4, 0, -1):
        with open(outfile, 'at+', encoding='utf8') as f:
            print(f"------precision even {i}------", file=f)
            test2(i, file=f)


if __name__ == "__main__":
    main()
